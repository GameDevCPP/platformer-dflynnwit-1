        GIT_TAG 2.6.x
)

    FetchContent_Populate(sfml)
    add_subdirectory(${sfml_SOURCE_DIR} ${sfml_BINARY_DIR})
endif()
get_cmake_property(_variableNames VARIABLES)
list(FILTER _variableNames INCLUDE REGEX "sfml")
